\begin{frame}{The litigator's problem}
  A case is won or lost on one question: \alert{does the evidence actually
  support what we pleaded?}
  \medskip
  \begin{itemize}
    \item A bundle is hundreds of pages: contracts, emails, witness
          statements, expert reports.
    \item Today a lawyer reads it all \textbf{by hand} to map every pleaded
          allegation to a document --- slow, late, and easy to miss.
    \item The worst outcome is an \alert{own-goal}: your \emph{own} bundle
          contains a document that contradicts your pleaded case --- found by
          the \emph{other side}, at trial.
  \end{itemize}
  \medskip
  \begin{block}{What a litigator needs, early}
    Spot weak, unsupported, or self-contradicting propositions \textbf{before}
    trial or settlement --- with every finding linked to its source for instant
    verification.
  \end{block}
\end{frame}

% ===========================================================================
\begin{frame}{The challenge: CMS $\times$ Harvey ``Pleading-to-Proof''}
  Take a litigation \textbf{bundle} + a \textbf{pleading} and stress-test the
  case theory:
  \begin{itemize}
    \item Map each pleaded allegation / defence to the evidence.
    \item Classify it \supported{} / \contradicted{} / \textbf{unaddressed}.
    \item Flag contradictions, gaps, and propositions needing more evidence.
    \item \alert{Keep the lawyer in control} --- the tool advises, the lawyer
          decides.
  \end{itemize}
  \vfill
  \begin{center}
    \footnotesize Prize: a CMS internship.\quad Our front end: \textbf{Pleading Compass}.
  \end{center}
\end{frame}

% ===========================================================================
\begin{frame}{How a litigator does it by hand (our north star)}
  From our team lawyer's analysis of the Meridian bundle:
  \begin{enumerate}
    \item Read the pleadings $\rightarrow$ recover the \emph{case theory}.
    \item Build a \textbf{chronology} of facts \& documents.
    \item Evaluate evidence: do the documents \emph{support} or
          \emph{contradict} the narrative?
    \item Weigh witness \& expert evidence (and admissibility).
    \item Resolve conflicts --- \emph{documents prevail over witnesses}.
    \item Re-evaluate the pleadings against what the evidence really shows.
  \end{enumerate}
  \medskip
  \begin{alertblock}{Our thesis}
    This rigorous-but-manual process is exactly what an engine can
    \textbf{systematise} --- without taking the judgement away from the lawyer.
  \end{alertblock}
\end{frame}

% ===========================================================================
\begin{frame}{Our method: ``A2'' --- a raw-documents $\rightarrow$ analysis engine}
  A systematic pipeline (\texttt{engine\_v2}):
  \begin{enumerate}
    \item \textbf{Split} every document into \alert{numbered paragraphs}
          --- enables clause-level retrieval + verbatim anchoring.
    \item \textbf{Extract} the pleaded propositions from the pleading.
    \item \textbf{Retrieve} the top-$K$ paragraphs per proposition (embeddings;
          instant local fallback).
    \item \textbf{Detect} contradictions: deterministic surfacers + an LLM
          classifier.
    \item \textbf{Verdict}: status + legal overlay + calibrated confidence +
          a \verify{} flag.
    \item \textbf{Ground} every quote \alert{verbatim} (anti-hallucination).
    \item \textbf{Emit} an \texttt{AppData} JSON the front end renders.
  \end{enumerate}
\end{frame}

\begin{frame}{Detecting contradictions: four exact surfacers + an LLM}
  Deterministic surfacers fire \emph{before} the model is trusted:
  \begin{itemize}
    \item \textbf{Numeric-disjoint} --- ``40\% of trading hours'' vs an expert's
          ``6.2\%''.
    \item \textbf{Supersession} --- a later Change Order overrides the date you
          sue on.
    \item \textbf{Contractual-allocation} --- a clause bars or caps the head of
          loss claimed.
    \item \textbf{Semantic-negation} --- ``we never accepted'' vs a signed
          acceptance certificate.
  \end{itemize}
  \medskip
  The LLM classifier then assesses anything subtler. Result $=$ a verdict with a
  \textbf{legal overlay}:
  \begin{center}
    \footnotesize
    \texttt{SUPERSEDED} $\cdot$ \texttt{CONTRACTUALLY\_BARRED} $\cdot$
    \texttt{CAPPED} $\cdot$ \texttt{CAUSATION\_PROBLEM} $\cdot$
    \texttt{BURDEN\_PROBLEM}
  \end{center}
\end{frame}

\begin{frame}{Anti-hallucination guarantees}
  The challenge cares about trust. We give four hard guarantees:
  \begin{enumerate}
    \item \textbf{Verbatim quotes only} --- every quote is a substring of its
          cited paragraph, or it is \alert{dropped}. Never invented.
    \item \textbf{Coverage guarantee} --- every pleaded paragraph gets an
          \emph{explicit} status. Blanks are impossible.
          \footnotesize(We caught \& fixed a silently unannotated paragraph,
          ``02\,\P6''.)\normalsize
    \item \textbf{Calibrated confidence + \verify{} flag} --- a blindly-trusting
          lawyer is, at worst, told ``verify this''. Never misled.
    \item \textbf{Deterministic surfacers} --- exact and reproducible, not a
          model's opinion.
  \end{enumerate}
\end{frame}

        \begin{itemize}\footnotesize
          \item semantic near-miss
          \item numeric / unit mismatch
          \item impossible chronology
          \item broken multi-doc inference chain
          \item superseded near-duplicate clause
          \item genuinely-ambiguous claim ($\Rightarrow$ \verify{})
          \item allocation / burden trap
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \medskip
  \begin{alertblock}{Automated harness --- the hard invariant}
    A planted-false claim is \textbf{never} marked \supported{} at high
    confidence --- \emph{and} the engine must not ``cry wolf'' on a clean
    control.
  \end{alertblock}
