# Backend dependencies for the graph pipeline + the mapper.
# The mapper, grounding and contradiction modules are pure standard library;
# these packages are what the pipeline itself needs to run a case end to end.

openai                    # OpenAI-protocol client (also the seam for a local/hosted Nemotron)
google-cloud-aiplatform   # Google Vertex AI client
vertexai                  # Vertex AI text embeddings
python-docx               # read .docx source documents in a case bundle
pydantic                  # config and schema models
pyyaml                    # configs/*.yaml and case manifest.yaml
